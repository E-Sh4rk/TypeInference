    \section{New typing system}

    \begin{align*}
      &\Gamma^p_{\Gamma,es,ts}=\fixpoint_\Gamma (\Gamma^p_{es,ts})\\&\\
      &\Gamma^p_{\varnothing,\varnothing} (\Gamma) = \Gamma\\
      &\Gamma^p_{\cons e {es},\cons t {ts}} (\Gamma) = \Gamma^p_{\Gamma,e,t},\Gamma^p_{es,ts}(\Gamma^p_{\Gamma,e,t})
    \end{align*}

    \section{New (intermediate) semantics}

    \[
      \begin{array}{lrcl}
      \textbf{Expressions} & e & ::= & c\alt x \alt \lambda^tx.e \alt \ite e t e e \alt e e\\
      \textbf{Values} & v & ::= & c \alt \lambda^tx.e
      \end{array}
    \]

    \[
      \begin{array}{lrcl}
      \textbf{Context} & C[] & ::= & e [] \alt [] v
      \end{array}
    \]

    \begin{mathpar}
      \Infer[Forget]
      {}
      {(e,\rho) \leadsto (e,\varnothing)}
      {}\\
      \Infer[Ctx]
      {(e,\varnothing) \leadsto (e',\rho)}
      {(C[e],\varnothing) \leadsto (C[e'],\rho)}
      {}\\
      \Infer[App]
      {}
      {((\lambda^tx.e) v, \varnothing) \leadsto (e\subst x v, \subst x v)}
      {}\\
      \Infer[TestCtx]
      {(e,\varnothing) \leadsto (e',\rho)}
      {(\ite e t {e_1} {e_2}, \varnothing) \leadsto (\ite {e'} t {e_1\rho} {e_2\rho}, \varnothing)}
      {}\\
      \Infer[Ite]
      {i \in \{1,2\}\hspace{1cm} (i=1 \Leftrightarrow v \in t)}
      {(\ite v t {e_1} {e_2},\varnothing) \leadsto (e_i,\varnothing)}
      {}\\
    \end{mathpar}
